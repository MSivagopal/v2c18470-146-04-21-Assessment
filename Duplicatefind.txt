package prac;

public class Duplicatefind {

	public static void main(String[] args) {
		String str = "Happy";
		char h ='p';
		int times = 0;
		for(int i=0; i<str.length();i++) 
		{
			if(str.charAt(i)==h)
			{
				times++;
			}
		
		}
		System.out.println("count of P is : "+times );
	}
}
